    \begin{frame}{Connections are central to the labor market}
        \begin{itemize}  [<+->]
            \item Hiring via connections benefit firms and workers alike
                \begin{itemize}
                    \item {40-50\% of all jobs found through social connections \footnotesize{\citep{topa2019social,granovetter1995getting}}}
                    \item {Higher hiring probability, lower turnover, and wage premiums for referred workers \footnotesize{\citep{brown2016informal,dustmann2016referral,friebel2023employee}}}
                \end{itemize}
            \item Network connections are not formed at random
                \begin{itemize}
                    \item {Networks exhibit homophily in social class, as people connect more often with similar others \footnotesize{\citep{mcpherson2001birds}}}
                    \item {Initial differences in average network employment status or education level can propagate inequality in labor market outcomes \footnotesize{\citep{calvo2004effects,calvo2009peer}}}
                    \item {Unequal access to valuable connections disadvantage Low-SES individuals \footnotesize{\citep{lin1981social,mouw2003social,chetty2021social}}}
                \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Referrals amplify network effects} 
                \begin{itemize} [<+->]
                    \item {Differential treatment within existing networks, e.g., referral of strong ties such as family and/or close friends \footnotesize{\citep{beaman2012gets,kramarz2014strong,hederos2025gender}}} 
                    \item {As well as biaseses against groups, e.g., race and gender \footnotesize{\citep{smith2005dont,ditomaso2013american,beaman2018job}}}
                    \item Both could play against Low-SES who have different networks to begin with
                \end{itemize}

    \end{frame}

\begin{frame}{Contribution}
        \begin{itemize}
            \item  Unique setup where we observe both an entire network and referral behaviors within that network, isolating the processes driving Low-SES inequality
            % \item Observing referral behaviors under two different incentive structures
        \end{itemize}
    \end{frame}

    \begin{frame}{Research Questions}
        \begin{itemize} [<+->]
            \item \textit{Could the class differences in labor market be driven by biases in referrals or by network structure?}
            \item \textit{Do network structures differ by social class?}
            \item \textit{Are there social class biases in referrals beyond the network structure?}
        \end{itemize}
    \end{frame}

        \begin{columns}[T]
            \begin{column}{0.38\textwidth}

                    \item Ask students to refer someone they have taken at least one course with
                    \item Observe the entire co-enrollment network at UNAB

    \begin{frame}{Is there a SES bias in referrals beyond networks?}
        
        \vspace{0.5em}
        \textbf{Conditional FE Logit:}
        \begin{align*}
            \Pr(\text{Refer}_{ij} = 1) = \Lambda\left(\beta_1 \text{SES}_j + \beta_2 \text{Courses}_j + \beta_3 \text{Score}_{ij} + \beta_4 \text{Courses}_j \times \text{Score}_{ij} + \alpha_i\right)
        \end{align*}
        
        \vspace{0.5em}
        \begin{itemize}% [<+->]
            \item $\text{Refer}_{ij}$: Binary outcome indicating whether individual $i$ refers individual $j$
            \item $\text{SES}_j$: Referral $j$ is Low, Middle, or High SES
            \item $\text{Courses}_{ij}$: Standardized number of courses taken together for $i$ and $j$
            \item $\text{Score}_j$: Standardized Math or Reading score of referral $j$
            \item $\alpha_i$: Individual fixed effect for referrer $i$
            \item Pool for each SES group
        \end{itemize}
    \end{frame}

\begin{frame}{Summary} 
        \begin{itemize}% [<+->]
        \item Networks are separated by SES
        \pause
        \item Referrers refer equally well across SES, and pick close ties with higher scores
        \pause
        \item Little to no bias in referrals in contrast to stark differences in network structures
    \end{itemize}
    \end{frame}

    \begin{frame}{Implications} 
        \markconclusion
        \begin{itemize} [<+->]
        \item Individuals across SES refer equally well with proper incentives and without bias
        \item Differences in network structures lie at the heart of the problem for solving inequality
    \end{itemize}
    \end{frame}
