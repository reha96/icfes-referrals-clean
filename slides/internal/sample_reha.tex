\begin{frame}{Empirical specification: referral decision}
    
    \vspace{0.5em}
    \textbf{Conditional Logit Model:}
    \begin{align*}
        \Pr(\text{Refer}_{ij} = 1) = \Lambda\left(\beta_1 \text{SES}_j + \beta_2 \text{Score}_j + \beta_3 \text{Tie}_{ij} + \beta_4 \text{Score}_j \times \text{Tie}_{ij} + \alpha_i\right)
    \end{align*}
    
    \vspace{0.5em}
    \begin{itemize}% [<+->]
        \item $\text{Refer}_{ij}$: Binary outcome indicating whether individual $i$ refers individual $j$
        \item $\text{SES}_j$: Referral $j$ is Low, Middle, or High SES
        \item $\text{Score}_j$: Math or Reading z-score of referral $j$
        \item $\text{Tie}_{ij}$: z-score of number of classes taken together for $i$ and $j$
        \item $\alpha_i$: Individual fixed effect for referrer $i$
    \end{itemize}
    
    \vspace{0.5em}
    \textbf{Features:}
    \begin{itemize}% [<+->]
        \item Control for referrer networks other unobservables with FE
        \item Estimate separately for Math and Reading scores
    \end{itemize}
\end{frame}
