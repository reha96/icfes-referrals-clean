\begin{frame}{Empirical specification}
    \begin{itemize}
        \item<1->{\only<1>{Model the referral choice using a mixed multinomial logit model for each referrer's choice set (average 175 potential nominees) [McFadden and Train, 2000]}\only<2->{{\color{grayedout}Model the referral choice using a mixed multinomial logit model for each referrer's choice set (average 175 potential nominees) [McFadden and Train, 2000]}}}
        
        \item<2->{\only<2>{The probability that student $i$ refers student $j \in N_i$ is given by
        $$\frac{e^{\beta_i x_{ij}}}{\sum_{j=1}^{N_i} e^{\beta_i x_{ij}}}$$
        
        where $x_{ij}$ is the vector of characteristics of $j$ (low-SES, test score, tie strength) and $\beta_i$ is a vector of individual-specific parameters comprising
        of a constant and a random component}}
    \end{itemize}
\end{frame}

\begin{frame}{Empirical Specification}
    
    \vspace{0.5em}
    \textbf{OLS Model:}
    \begin{align*}
        \text{Score}_{ij} = \beta_0 &+ \beta_1 \cdot \text{X}_i + \beta_2 \cdot \text{Mean Score}_j + \beta_3 \cdot \text{SD Score}_j + \varepsilon_{ij}
    \end{align*}
    
    \vspace{0.5em}
    \begin{itemize}
        \item<1->{\only<1>{$\text{Score}_{ij}$: Test score of the nominee from referrer $i$ in subject $j$}\only<2->{{\color{grayedout}$\text{Score}_{ij}$: Test score of the nominee from referrer $i$ in subject $j$}}}
        \item<2->{\only<2>{$\text{X}_{ij}$: SES of student $i$ or the test score of student $i$ in subject $j$}\only<3->{{\color{grayedout}$\text{X}_{ij}$: SES of student $i$ or the test score of student $i$ in subject $j$}}}
        \item<3->{\only<3>{$\text{Mean Score}_j$ and $\text{SD Score}_j$: Mean and SD of $i$'s network performance in subject $j$}\only<4->{{\color{grayedout}$\text{Mean Score}_j$ and $\text{SD Score}_j$: Mean and SD of $i$'s network performance in subject $j$}}}
    \end{itemize}
    
    \vspace{0.5em}
    \textbf{Features:}
