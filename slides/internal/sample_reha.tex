    \begin{frame}{Is there a SES bias in referrals?}
        
        \vspace{0.5em}
        \textbf{Conditional Logit Model:}
        \begin{align*}
            \Pr(\text{Refer}_{ij} = 1) = \Lambda\left(\beta_1 \text{SES}_j + \beta_2 \text{Score}_j + \beta_3 \text{Tie}_{ij} + \beta_4 \text{Score}_j \times \text{Tie}_{ij} + \alpha_i\right)
        \end{align*}
        
        \vspace{0.5em}
        \begin{itemize}% [<+->]
            \item $\text{Refer}_{ij}$: Binary outcome indicating whether individual $i$ refers individual $j$
            \item $\text{SES}_j$: Referral $j$ is Low, Middle, or High SES
            \item $\text{Score}_j$: Standardized Math or Reading score of referral $j$
            \item $\text{Tie}_{ij}$: Standardized number of classes taken together for $i$ and $j$
            \item $\alpha_i$: Individual fixed effect for referrer $i$
        \end{itemize}
        
        \vspace{0.5em}
        \textbf{Features:}
        \begin{itemize}% [<+->]
            \item FE controls for referrer network structure and other unobservables
        \end{itemize}
    \end{frame}

    \begin{frame}[label=good_refferals]{Who makes better referrals?}
    
        \vspace{0.5em}
        \textbf{OLS Model:}
        \begin{align*}
            \text{Premium}_{ij} = \beta_0 + \beta_1 \text{SES}_i + \beta_2 \text{Score}_i + \beta_3 \Delta\text{OBlf}_i + \beta_4 \Delta\text{NBlf}_i + \text{X}_i'\boldsymbol{\gamma} + \epsilon_{i}
        \end{align*}
        
        \vspace{0.5em}
        \begin{itemize}% [<+->]
            \item $\text{Premium}_{i}$: Nominee $j$'s test z-score minus mean score of $i$'s network
            \item $\text{SES}_{i}$: Referrer $i$'s socioeconomic status (Low, Middle, High)    
            \item $\Delta\text{OBlf}_i$, $\Delta\text{NBlf}_i$: $i$'s beliefs on own and nominee test scores minus actual scores (standardized)
            \item $\text{Score}_i$: Referrer $i$'s own test z-score 
        \end{itemize}
    
        \vspace{0.5em}
        \textbf{Controls:}
        \begin{itemize}% [<+->]
                \item Referrer $i$'s treatment (Baseline vs. Bonus)    
                \item Test area indicator (Math vs. Reading)
                \item Number of classes taken together for $i$ and nominee $j$
        \end{itemize}
    \end{frame}
