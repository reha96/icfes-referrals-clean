    \begin{frame}{Motivation and Research Question}
        \showprogressbar
        \begin{itemize} [<+->]
            \item Understand persistent class differences in labor the market, like the underrepresentation of Low-SES researchers in top academic institutions [Stansbury and Rodriguez, 2024]
            \item Focus on class biases in referrals
            \begin{itemize}
                \item Most jobs are found through referrals [Topa, 2011]
                \item Referals depend on social networks, which exhibit homophily [McPherson et al., 2001]
            \end{itemize}
            \item Are there social class biases in referrals?
            \item If so, what are the potential drivers of these biases?
        \end{itemize}
    \end{frame}

        \begin{columns}[T]

                    \item Approx. 6000 students across all social classes
                    \item Administrative data including gender, age,

    \end{frame}

    \begin{frame}{Design}
        \begin{columns}[T]
            \begin{column}{0.38\textwidth}
                \begin{itemize} [<+->]
                    \item Ask students to refer someone they have taken at least one class with
                    \item Pay according to the student's math and verbal scores in the national entry exam
                    \item Incentivize better refferals by increasing reward as referral score goes higher 
                \end{itemize}
            \end{column}
            \begin{column}{0.62\textwidth}
                \centering
                \includegraphics[width=0.8\columnwidth]{/Users/reha.tuncer/Documents/GitHub/icfes-referrals/figures/interface.png}
            \end{column}
        \end{columns}
    \end{frame}


    \begin{frame}{Procedures}
        \begin{itemize} [<+->]
            \item Recruited participants by emailing 4500 students ($>$1st year)
            \item 30 minute online experiment in Qualtrics
            \item Average payment of 80 USD (lottery for 1 out of 10 participants)
            \item 840 complete responses
            \item Final sample 734 participants who referred someone they took a class with
        \end{itemize}
    \end{frame}
